**Step 5 — main.rs: Handle new Result types from format functions**

Verify arm — wrap `format_verify_json` call in match:
```rust
Commands::Verify { path } => {
    match verify::run_verify(path) {
        Ok(result) => {
            if cli.json {
                match verify::format_verify_json(&result, path) {
                    Ok(json) => print!("{}", json),
                    Err(e) => {
                        eprintln!("Error: {e}");
                        std::process::exit(1);  // JSON format error = logic failure, NOT path-not-found
                    }
                }
            } else {
                print!("{}", verify::format_verify_human(&result, path));
            }
            if result.valid {
                std::process::exit(0);
            } else {
                std::process::exit(1);
            }
        }
        Err(e) => {
            eprintln!("Error: {e}");
            std::process::exit(2);  // Bundle dir not found
        }
    }
}
```

Inspect arm — wrap `format_inspect_json` call in match:
```rust
Commands::Inspect { path } => {
    match inspect::run_inspect(path) {
        Ok(manifest) => {
            if cli.json {
                match inspect::format_inspect_json(&manifest) {
                    Ok(json) => print!("{}", json),
                    Err(e) => {
                        eprintln!("Error: {e}");
                        std::process::exit(1);  // JSON format error = logic failure
                    }
                }
            } else {
                print!("{}", inspect::format_inspect_human(&manifest, path));
            }
            std::process::exit(0);
        }
        Err(e) => {
            eprintln!("Error: {e}");
            match e {
                crate::error::BundleError::ManifestNotFound(_) => std::process::exit(2),
                _ => std::process::exit(1),
            }
        }
    }
}
```

CRITICAL: The `Err(e)` from `format_inspect_json`/`format_verify_json` must use `exit(1)` — NOT `exit(2)`. Exit 2 is reserved for "path not found" (from `run_verify`/`run_inspect` Err). A JSON serialization failure during formatting is a logic error (exit 1).

**Avoid:**
- Do NOT add `.unwrap()` at any call site in main.rs — that is the anti-pattern being eliminated
- Do NOT route JsonSerialize errors through the outer Err(e) arm — they must be handled inline after the Ok arm succeeds
- Do NOT change create.rs line 244 to use `?` — the json! literal is infallible and the code path exits unconditionally
